namespace block_mycourse_recommendations;

defined('MOODLE_INTERNAL') || die();

/**
 * The class that encapsulates the query made to database (database_helper->query_data()). This class will be
 * used in those classes which will need to access the queried data.
 *
 * @package    block_mycourse_recommendations
 * @copyright  2016 onwards Julen Pardo & Mondragon Unibertsitatea
 * @license    http://www.gnu.org/copyleft/gpl.html GNU GPL v3 or later
 */
class query_result {

    /**
     * User id.
     *
     * @var int
     */
    private $userid;

    /**
     * Course id.
     *
     * @var int
     */
    private $courseid;

    /**
     * Module id.
     *
     * @var int
     */
    private $moduleid;

    /**
     * Module name.
     *
     * @var string
     */
    private $modulename;

    /**
     * Module's log views.
     *
     * @var int
     */
    private $logviews;

    /**
     * Module's grade.
     *
     * @var float
     */
    private $grades;

    /**
     * Instantiates the class, receiving a value for each queried field.
     *
     * @param int $userid
     * @param int $courseid
     * @param int $moduleid
     * @param string $modulename
     * @param int $logviews
     * @param float $grades Default set to -1, because it can be empty in database.
     */
    public function __construct($userid, $courseid, $moduleid, $modulename, $logviews, $grades = -1) {
        $this->userid = $userid;
        $this->courseid = $courseid;
        $this->moduleid = $moduleid;
        $this->modulename = $modulename;
        $this->logviews = $logviews;
        $this->grades = $grades;
    }

    /**
     * Returns user id.
     *
     * @return int
     */
    public function get_userid() {
        return $this->userid;
    }

    /**
     * Returns course id.
     *
     * @return int
     */
    public function get_courseid() {
        return $this->moduleid;
    }

    /**
     * Returns module id.
     *
     * @return int
     */
    public function get_moduleid() {
        return $this->moduleid;
    }

    /**
     * Returns module name.
     *
     * @return string
     */
    public function get_modulename() {
        return $this->modulename;
    }

    /**
     * Returns log views.
     *
     * @return int
     */
    public function get_logviews() {
        return $this->logviews;
    }

    /**
     * Returns grade.
     *
     * @return float
     */
    public function get_grades() {
        return $this->grades;
    }
}
